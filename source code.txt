### 1. `app/_layout.tsx` — Root layout + font loader

```tsx
import { Stack } from "expo-router";
import { StatusBar } from "expo-status-bar";
import { GestureHandlerRootView } from "react-native-gesture-handler";
import { View, ActivityIndicator, StyleSheet } from "react-native";
import {
  useFonts as usePlayfair,
  PlayfairDisplay_700Bold,
  PlayfairDisplay_400Regular_Italic,
} from "@expo-google-fonts/playfair-display";
import {
  useFonts as useOutfit,
  Outfit_400Regular,
  Outfit_600SemiBold,
} from "@expo-google-fonts/outfit";
import {
  useFonts as useCaveat,
  Caveat_400Regular,
  Caveat_700Bold,
} from "@expo-google-fonts/caveat";

export default function RootLayout() {
  const [playfairLoaded] = usePlayfair({
    PlayfairDisplay_700Bold,
    PlayfairDisplay_400Regular_Italic,
  });
  const [outfitLoaded] = useOutfit({
    Outfit_400Regular,
    Outfit_600SemiBold,
  });
  const [caveatLoaded] = useCaveat({
    Caveat_400Regular,
    Caveat_700Bold,
  });

  const ready = playfairLoaded && outfitLoaded && caveatLoaded;

  if (!ready) {
    return (
      <View style={styles.loader}>
        <ActivityIndicator color="#D4AF37" size="large" />
      </View>
    );
  }

  return (
    <GestureHandlerRootView style={{ flex: 1, backgroundColor: "#1A0B1A" }}>
      <StatusBar style="light" />
      <Stack
        screenOptions={{
          headerShown: false,
          contentStyle: { backgroundColor: "#1A0B1A" },
          animation: "fade",
        }}
      />
    </GestureHandlerRootView>
  );
}

const styles = StyleSheet.create({
  loader: {
    flex: 1,
    backgroundColor: "#1A0B1A",
    alignItems: "center",
    justifyContent: "center",
  },
});
```

### 3. `src/components/FloatingParticles.tsx` — Reusable particle layer

```tsx
import React, { useEffect, useMemo } from "react";
import { StyleSheet, View, Dimensions } from "react-native";
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withRepeat,
  withTiming,
  withDelay,
  Easing,
  interpolate,
} from "react-native-reanimated";
import { Heart, Sparkles, Star } from "lucide-react-native";

const { width, height } = Dimensions.get("window");

type ParticleProps = { index: number; total: number };

function Particle({ index, total }: ParticleProps) {
  const progress = useSharedValue(0);

  const startX = useMemo(() => Math.random() * width, []);
  const drift = useMemo(() => (Math.random() - 0.5) * 80, []);
  const duration = useMemo(() => 6000 + Math.random() * 5000, []);
  const delay = useMemo(() => (index / total) * duration, [index, total, duration]);
  const size = useMemo(() => 14 + Math.random() * 18, []);
  const Icon = useMemo(() => {
    const choices = [Heart, Sparkles, Star];
    return choices[index % choices.length];
  }, [index]);
  const color = useMemo(() => {
    const palette = ["#D4AF37", "#E0BFB8", "#9E4770", "#F8F0E3"];
    return palette[index % palette.length];
  }, [index]);

  useEffect(() => {
    progress.value = withDelay(
      delay,
      withRepeat(withTiming(1, { duration, easing: Easing.linear }), -1, false)
    );
  }, [progress, duration, delay]);

  const animatedStyle = useAnimatedStyle(() => {
    const translateY = interpolate(progress.value, [0, 1], [height + 40, -60]);
    const translateX = interpolate(progress.value, [0, 0.5, 1], [0, drift, 0]);
    const opacity = interpolate(progress.value, [0, 0.15, 0.85, 1], [0, 0.9, 0.9, 0]);
    const scale = interpolate(progress.value, [0, 0.5, 1], [0.6, 1, 0.7]);
    const rotate = `${interpolate(progress.value, [0, 1], [0, 360])}deg`;
    return {
      transform: [{ translateY }, { translateX }, { scale }, { rotate }],
      opacity,
    };
  });

  return (
    <Animated.View
      pointerEvents="none"
      style={[styles.particle, { left: startX }, animatedStyle]}
    >
      <Icon size={size} color={color} fill={color} />
    </Animated.View>
  );
}

export default function FloatingParticles({ count = 14 }: { count?: number }) {
  const particles = Array.from({ length: count }, (_, i) => i);
  return (
    <View pointerEvents="none" style={StyleSheet.absoluteFillObject}>
      {particles.map((i) => (
        <Particle key={i} index={i} total={count} />
      ))}
    </View>
  );
}

const styles = StyleSheet.create({
  particle: { position: "absolute", top: 0 },
});
```

```tsx
  const nameProgress = useSharedValue(0);
  const ctaProgress = useSharedValue(0);
  const glowProgress = useSharedValue(0);
  const giftFloat = useSharedValue(0);

  useEffect(() => {
    titleProgress.value = withDelay(200, withTiming(1, { duration: 900, easing: Easing.out(Easing.cubic) }));
    nameProgress.value = withDelay(900, withTiming(1, { duration: 1100, easing: Easing.out(Easing.cubic) }));

    ctaProgress.value = withDelay(2400, withTiming(1, { duration: 700 }));

      withSequence(
        withTiming(1, { duration: 2200, easing: Easing.inOut(Easing.ease) }),
        withTiming(0, { duration: 2200, easing: Easing.inOut(Easing.ease) })
      ), -1, false);
    giftFloat.value = withRepeat(
      withSequence(
        withTiming(1, { duration: 1800, easing: Easing.inOut(Easing.ease) }),
        withTiming(0, { duration: 1800, easing: Easing.inOut(Easing.ease) })
      ), -1, false);
  }, [titleProgress, nameProgress, subProgress, ctaProgress, glowProgress, giftFloat]);

    opacity: titleProgress.value,

        />
      </Animated.View>
      <FloatingParticles count={16} />

        <Animated.View style={[styles.giftWrap, giftStyle]}>
          <View style={styles.giftRing}>
```

### 5. `app/gallery.tsx` — 3D Polaroid Gallery

```tsx
import React from "react";
import { StyleSheet, View, Text, Dimensions, Pressable } from "react-native";
import { LinearGradient } from "expo-linear-gradient";
import { useRouter } from "expo-router";
import Animated, {
  useSharedValue, useAnimatedScrollHandler, useAnimatedStyle,
  interpolate, Extrapolation,
} from "react-native-reanimated";
import { Image } from "expo-image";
import { ArrowLeft, Hand, Heart } from "lucide-react-native";
import FloatingParticles from "../src/components/FloatingParticles";
import { memories, Memory } from "../src/data/memories";

const { width } = Dimensions.get("window");
const CARD_WIDTH = Math.min(width * 0.72, 320);
const CARD_HEIGHT = CARD_WIDTH * 1.32;
const SPACING = 20;
const SNAP_INTERVAL = CARD_WIDTH + SPACING;
const SIDE_PADDING = (width - CARD_WIDTH) / 2;

function PolaroidCard({ item, index, scrollX, onPress }: {
  item: Memory; index: number;
  scrollX: Animated.SharedValue<number>; onPress: () => void;
}) {
  const animatedStyle = useAnimatedStyle(() => {
    const input = [(index - 1) * SNAP_INTERVAL, index * SNAP_INTERVAL, (index + 1) * SNAP_INTERVAL];
    const scale = interpolate(scrollX.value, input, [0.82, 1, 0.82], Extrapolation.CLAMP);
    const rotateY = interpolate(scrollX.value, input, [25, 0, -25], Extrapolation.CLAMP);
    const translateY = interpolate(scrollX.value, input, [30, 0, 30], Extrapolation.CLAMP);
    const opacity = interpolate(scrollX.value, input, [0.55, 1, 0.55], Extrapolation.CLAMP);
    return {
      opacity,
      transform: [
        { perspective: 900 }, { translateY }, { scale }, { rotateY: `${rotateY}deg` },
      ],
    };
  });
  const tiltStyle = useAnimatedStyle(() => {
    const input = [(index - 1) * SNAP_INTERVAL, index * SNAP_INTERVAL, (index + 1) * SNAP_INTERVAL];
    const rotateZ = interpolate(scrollX.value, input, [-3, 0, 3], Extrapolation.CLAMP);
    return { transform: [{ rotate: `${rotateZ}deg` }] };
  });

  return (
    <Animated.View style={[styles.cardOuter, animatedStyle]}>
      <Pressable onPress={onPress} style={styles.pressArea}>
        <Animated.View style={[styles.polaroid, tiltStyle]}>
          <View style={styles.photoFrame}>
            <Image source={{ uri: item.image }} style={styles.photo} contentFit="cover" transition={400} />
            <LinearGradient colors={["transparent", "rgba(26,11,26,0.7)"]} style={styles.photoShade} />
            <View style={styles.dateChip}>
              <Text style={styles.dateChipText}>{item.date}</Text>
            </View>
          </View>
          <View style={styles.polaroidCaption}>
            <Text style={styles.captionTitle} numberOfLines={1}>{item.title}</Text>
            <View style={styles.captionRow}>
              <Hand size={12} color="#D4AF37" />
              <Text style={styles.captionHint}>tap to unfold</Text>
            </View>
          </View>
        </Animated.View>
      </Pressable>
    </Animated.View>
  );
}

export default function Gallery() {
  const router = useRouter();
  const scrollX = useSharedValue(0);
  const onScroll = useAnimatedScrollHandler((e) => { scrollX.value = e.contentOffset.x; });

  return (
    <View style={styles.container}>
      <LinearGradient colors={["#1A0B1A", "#2A1528", "#1A0B1A"]} style={StyleSheet.absoluteFillObject} />
      <FloatingParticles count={10} />
      <View style={styles.header}>
        <Pressable onPress={() => router.back()} style={styles.iconBtn}>
          <ArrowLeft size={20} color="#F8F0E3" />
        </Pressable>
        <View style={styles.headerCenter}>
          <Text style={styles.headerEyebrow}>For Priyanka</Text>
          <Text style={styles.headerTitle}>Memory Lane</Text>
        </View>
        <View style={styles.iconBtn}><Heart size={18} color="#D4AF37" fill="#D4AF37" /></View>
      </View>
      <View style={styles.carouselWrap}>
        <Animated.ScrollView
          horizontal showsHorizontalScrollIndicator={false} decelerationRate="fast"
          snapToInterval={SNAP_INTERVAL} snapToAlignment="start"
          contentContainerStyle={{ paddingHorizontal: SIDE_PADDING }}
          onScroll={onScroll} scrollEventThrottle={16}
        >
          {memories.map((m, i) => (
            <View key={m.id} style={{ width: CARD_WIDTH, marginRight: i === memories.length - 1 ? 0 : SPACING }}>
              <PolaroidCard item={m} index={i} scrollX={scrollX} onPress={() => router.push(`/memory/${m.id}`)} />
            </View>
          ))}
        </Animated.ScrollView>
      </View>
      <View style={styles.bottomDock}>
        <Text style={styles.dockHint}>Swipe through · Tap to reveal</Text>
        <Pressable onPress={() => router.push("/message")} style={styles.finalBtn}>
          <LinearGradient
            colors={["rgba(212,175,55,0.18)", "rgba(158,71,112,0.18)"]}
            start={{ x: 0, y: 0 }} end={{ x: 1, y: 1 }} style={styles.finalGradient}
          >
            <Text style={styles.finalText}>Read my letter to you</Text>
          </LinearGradient>
        </Pressable>
      </View>
    </View>
  );
}

const styles = StyleSheet.create({

  headerCenter: { alignItems: "center" },
  headerEyebrow: { fontFamily: "Outfit_400Regular", color: "#DAB6C4", fontSize: 11, letterSpacing: 3, textTransform: "uppercase", marginBottom: 2 },
  headerTitle: { fontFamily: "PlayfairDisplay_700Bold", color: "#F8F0E3", fontSize: 22, letterSpacing: 0.3 },
  carouselWrap: { flex: 1, justifyContent: "center" },
  cardOuter: { width: CARD_WIDTH, height: CARD_HEIGHT },
  pressArea: { flex: 1 },
  polaroid: {
    flex: 1, backgroundColor: "#F8F0E3", borderRadius: 6, padding: 12, paddingBottom: 22,
```

```tsx
import { useLocalSearchParams, useRouter } from "expo-router";

  useSharedValue, useAnimatedStyle, withTiming,
  interpolate, Easing,
} from "react-native-reanimated";
import { Image } from "expo-image";
import { ArrowLeft, RotateCw, Heart } from "lucide-react-native";
import FloatingParticles from "../../src/components/FloatingParticles";

const { width } = Dimensions.get("window");
const CARD_WIDTH = Math.min(width * 0.82, 340);
const CARD_HEIGHT = CARD_WIDTH * 1.35;

export default function MemoryDetail() {
  const { id } = useLocalSearchParams<{ id: string }>();
  const router = useRouter();
  const memory = memories.find((m) => m.id === id) ?? memories[0];
  const flip = useSharedValue(0);
  const enter = useSharedValue(0);

  useEffect(() => {
    enter.value = withTiming(1, { duration: 700, easing: Easing.out(Easing.cubic) });
  }, [enter]);

  const toggleFlip = () => {
    flip.value = withTiming(flip.value > 0.5 ? 0 : 1, { duration: 900, easing: Easing.inOut(Easing.cubic) });
  };

  const frontStyle = useAnimatedStyle(() => {
    const rotateY = `${interpolate(flip.value, [0, 1], [0, 180])}deg`;
    const opacity = flip.value < 0.5 ? 1 : 0;
    const scale = interpolate(enter.value, [0, 1], [0.85, 1]);
    return { opacity, transform: [{ perspective: 1200 }, { scale }, { rotateY }] };
  });
  const backStyle = useAnimatedStyle(() => {
    const rotateY = `${interpolate(flip.value, [0, 1], [180, 360])}deg`;
    const opacity = flip.value > 0.5 ? 1 : 0;
    const scale = interpolate(enter.value, [0, 1], [0.85, 1]);
    return { opacity, transform: [{ perspective: 1200 }, { scale }, { rotateY }] };
  });
  const fadeIn = useAnimatedStyle(() => ({
    opacity: enter.value,
    transform: [{ translateY: interpolate(enter.value, [0, 1], [12, 0]) }],
  }));

  return (
    <View style={styles.container}>
      <LinearGradient colors={["#1A0B1A", "#2A1528", "#1A0B1A"]} style={StyleSheet.absoluteFillObject} />
      <FloatingParticles count={8} />
      <Animated.View style={[styles.header, fadeIn]}>
        <Pressable onPress={() => router.back()} style={styles.iconBtn}>
          <ArrowLeft size={20} color="#F8F0E3" />
        </Pressable>
        <View style={styles.headerCenter}>
          <Text style={styles.headerEyebrow}>{memory.date}</Text>
          <Text style={styles.headerTitle}>{memory.title}</Text>
        </View>
        <View style={styles.iconBtn}><Heart size={18} color="#D4AF37" fill="#D4AF37" /></View>
      </Animated.View>
      <View style={styles.center}>
        <Pressable onPress={toggleFlip} style={styles.cardPress}>
          <View style={styles.cardStack}>
            <Animated.View style={[styles.cardFace, styles.cardFront, frontStyle]}>
              <View style={styles.photoFrame}>
                <Image source={{ uri: memory.image }} style={styles.photo} contentFit="cover" transition={500} />
                <LinearGradient colors={["transparent", "rgba(26,11,26,0.85)"]} style={styles.photoShade} />
                <View style={styles.tapHint}>
                  <RotateCw size={14} color="#1A0B1A" />
                  <Text style={styles.tapHintText}>Tap to read</Text>
                </View>
              </View>
              <Text style={styles.frontCaption}>{memory.title}</Text>
            </Animated.View>
            <Animated.View style={[styles.cardFace, styles.cardBack, backStyle]}>
              <LinearGradient colors={["#F8F0E3", "#E8DCC8"]} style={StyleSheet.absoluteFillObject} />
```
